pipeline {
agent any
       stages{
        stage('Clone'){
             steps{  
                 git branch: 'main', url: 'https://github.com/GSellami/Devops.git'
             }
        }
       
        stage('MVN CLEAN') {
            steps {
                sh 'mvn clean'
                 
            }
        }
           stage("Build") {
                 steps {
sh "mvn compile"
}}






   }
}
